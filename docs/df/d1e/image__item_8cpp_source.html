58ImageItem::ImageItem(VAbstractPattern *doc, DraftImage image, QGraphicsItem *parent)
59 : QGraphicsItem(parent)
60 , m_doc(doc)
61 , m_offset(QPointF(0.0, 0.0))
62 , m_boundingRect(QRectF())
63 , m_handleRect(QRectF())
64 , m_actualRect(QRectF())
65 , m_resizeHandles()
66 , m_resizePosition()
67 , m_rotateLine()
68 , m_angleHandle()
69 , m_angle()
70 , m_mousePressed(false)
71 , m_isHovered(false)
72 , m_selectionType(SelectionType::ByMouseRelease)
73 , m_transformationMode(Qt::SmoothTransformation)
74 , m_image(image)
75 , m_pixmapWidth()
76 , m_pixmapHeight()
77 , m_selectable(true)
78 , m_minDimension(16)
79 , m_maxDimension(60000)
80{
81 initializeItem();
82
83 m_boundingRect = QRectF(m_image.xPos, m_image.yPos, m_image.xPos + m_image.width, m_image.yPos + m_image.height);
84 m_handleRect = m_boundingRect.adjusted(HANDLE_SIZE/2, HANDLE_SIZE/2, -HANDLE_SIZE/2, -HANDLE_SIZE/2);
85
86 if (m_image.order == 0)
87 {
88 qreal minZValue = maxImageZvalue+1;
89 foreach (ImageItem *item, m_doc->getBackgroundImageMap().values())
90 {
91 minZValue = qMin(minZValue, item->m_image.order);
92 }
93 m_image.order = minZValue-1;
94 moveToTop();
95 }
96
97 updateImage();
98
99 m_resizeHandles = new ResizeHandlesItem(this, m_minDimension, m_maxDimension);
100 m_resizeHandles->setLockAspectRatio(m_image.aspectLocked);
101 m_resizeHandles->setParentRotation(m_image.rotation);
102 m_resizeHandles->parentIsLocked(m_image.locked);
103 m_resizeHandles->setVisible(m_image.locked);
104 connect(m_resizeHandles, &ResizeHandlesItem::sizeChanged, this, &ImageItem::updateFromHandles);
105 connect(m_resizeHandles, &ResizeHandlesItem::setStatusMessage, this, [this](QString message) {emit setStatusMessage(message);});
106}

181void ImageItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
182{
183 Q_UNUSED(option)
184 Q_UNUSED(widget)
185
186 if (isSelected())
187 {
188 painter->save();
189 painter->setPen(QPen(Qt::black, 1, Qt::DashLine));
190 painter->drawRect(m_boundingRect);
191 painter->restore();
192 }
193
194 if (!m_image.locked && m_isHovered)
195 {
196 painter->save();
197 QColor color = QColor(qApp->Settings()->getTertiarySupportColor());
198 color.setAlpha(20);
199 painter->setPen(QPen(Qt::black, 2, Qt::DashLine));
200 painter->setBrush(color);
201 painter->drawRect(m_boundingRect);
202 painter->restore();
203 }
204
205 painter->setRenderHint(QPainter::SmoothPixmapTransform, (m_transformationMode == Qt::SmoothTransformation));
206 painter->drawPixmap(m_boundingRect.x(), m_boundingRect.y(), m_image.width, m_image.height, m_image.pixmap);
207}

333 actionSeparator = new QAction(this);
334 actionSeparator->setSeparator(true);
335 menu.addAction(actionSeparator);
336
337 QAction *actionDelete = menu.addAction(QIcon("://icon/32x32/trashcan.png"), tr("Delete"));

391void ImageItem::mousePressEvent(QGraphicsSceneMouseEvent *event)
392{
393 if (!m_selectable)
394 {
395 event->ignore();
396 return;
397 }
398
399 // Special for not selectable item first need to call standard mousePressEvent then accept event
400 //QGraphicsItem::mousePressEvent(event);
401
402 // Somehow clicking on non selectable object does not clear previous selections.
403 if (not (flags() & ItemIsSelectable) && scene())
404 {
405 scene()->clearSelection();
406 }
407 m_mousePressed = true;
408
409 if (flags() & QGraphicsItem::ItemIsMovable)
410 {
411 if (event->button() == Qt::LeftButton && event->type() != QEvent::GraphicsSceneMouseDoubleClick)
412 {
413 SetItemOverrideCursor(this, cursorArrowCloseHand, 1, 1);
414 m_offset = event->pos() - m_boundingRect.topLeft();
415 }
416 }
417 if (m_selectionType == SelectionType::ByMouseRelease)
418 {
419 event->accept(); // This help for non selectable items still receive mouseReleaseEvent events
420 }
421 else
422 {
423 if (event->button() == Qt::LeftButton && event->type() != QEvent::GraphicsSceneMouseDoubleClick)
424 {
425 emit imageSelected(m_image.id);
426 event->accept();
427 }
428 }
429}

527void ImageItem::moveToBottom()
528{
529 qreal minZValue = m_image.order;
530 foreach (ImageItem *item, m_doc->getBackgroundImageMap().values())
531 {
532 if (item != this && item->m_image.order < m_image.order)
533 {
534 minZValue = qMin(minZValue, item->m_image.order);
535 item->m_image.order++;
536 item->updateImage();
537 }
538 }
539 m_image.order = minZValue;
540 updateImage();
541}

544void ImageItem::moveToTop()
545{
546 foreach (ImageItem *item, m_doc->getBackgroundImageMap().values())
547 {
548 if (item != this && item->m_image.order > m_image.order)
549 {
550 item->m_image.order--;
551 item->updateImage();
552 }
553 }
554 m_image.order = maxImageZvalue;
555 updateImage();
556}

558void ImageItem::moveUp()
559{
560 if (m_image.order == maxImageZvalue)
561 {
562 return;
563 }
564
565 foreach (ImageItem *item, m_doc->getBackgroundImageMap().values())
566 {
567 if (item->m_image.order == m_image.order + 1)
568 {
569 item->m_image.order--;
570 item->updateImage();
571 }
572 }
573 m_image.order ++;
574 updateImage();
575}

578void ImageItem::moveDown()
579{
580 if (m_image.order == maxImageZvalue-m_doc->getBackgroundImageMap().values().size()+1)
581 {
582 return;
583 }
584
585 foreach (ImageItem *item, m_doc->getBackgroundImageMap().values())
586 {
587 if (item->m_image.order == m_image.order - 1)
588 {
589 item->m_image.order++;
590 item->updateImage();
591 }
592 }
593 m_image.order --;
594 updateImage();
595}

598void ImageItem::showImageStatusMessage()
599{

605 if (m_image.units == Unit::Px)

623 .arg(height)

630 emit setStatusMessage(message);
